119 sl_status_t wf200_apb_read_32( uint32_t address, uint32_t* value_out )
120 {
121  uint32_t value32;
122  sl_status_t result;
123 
124  // write address
125  result = wf200_reg_write_32( WF200_SRAM_BASE_ADDR_REG_ID, address );
126 
127  // set the "prefetch" bit
128  result = wf200_reg_read_32( WF200_CONFIG_REG_ID, &value32 );
129  result = wf200_reg_write_32( WF200_CONFIG_REG_ID, value32 | CONFIG_PREFECH_BIT );
130 
131  // and wait for the prefetch bit to clear
132  for ( uint32_t i = 0; i < 20; i++ )
133  {
134  result = wf200_reg_read_32( WF200_CONFIG_REG_ID, &value32 );
135  if ( ( value32 & CONFIG_PREFECH_BIT ) == 0 )
136  {
137  break;
138  }
139 
140  wf200_host_wait( 1 );
141  }
142 
143  // and data is ready
144  result = wf200_reg_read_32( WF200_SRAM_DPORT_REG_ID, value_out );
145 
146  return result;
147 }
