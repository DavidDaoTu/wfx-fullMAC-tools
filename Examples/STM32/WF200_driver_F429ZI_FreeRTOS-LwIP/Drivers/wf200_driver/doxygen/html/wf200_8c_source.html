95 sl_status_t wf200_init( wf200_context_t* context )
96 {
97  sl_status_t result;
98  HiStartupInd_t* startup_info;
99 
100  memset( context, 0, sizeof( *context ) );
101 
102  wf200_context = context;
103  wf200_context->used_buffer_number = 0;
104 
105  result = wf200_init_bus( );
106  ERROR_CHECK( result );
107 #ifdef DEBUG
108  printf("--Bus initialized--\r\n");
109 #endif
110 
111  result = wf200_init_chip( );
112  ERROR_CHECK( result );
113 #ifdef DEBUG
114  printf("--Chip initialized--\r\n");
115 #endif
116 
117  result = wf200_download_run_bootloader();
118  ERROR_CHECK( result );
119 #ifdef DEBUG
120  printf("--Bootloader downloaded--\r\n");
121 #endif
122 
123  wf200_context->waited_event_id = HI_STARTUP_IND_ID;
124 
125  /* Downloading wf200 firmware contained in wfm_wf200.h */
126  result = wf200_download_run_firmware( );
127  ERROR_CHECK( result );
128 #ifdef DEBUG
129  printf("--Firmware downloaded--\r\n");
130 #endif
131 
132  result = wf200_enable_irq( );
133  ERROR_CHECK( result );
134 
135  result = wf200_set_access_mode_message( );
136  ERROR_CHECK( result );
137 #ifdef DEBUG
138  printf("--Message mode set--\r\n");
139 #endif
140 
141  /* Waiting for the startup indication from wf200, HI_STARTUP_IND_ID */
142  result = wf200_host_wait_for_confirmation( WF200_DEFAULT_REQUEST_TIMEOUT, (void **)&startup_info );
143  ERROR_CHECK( result );
144 
145  /* Storing mac addresses from wf200 in the context */
146  memcpy(&(context->mac_addr_0.octet), startup_info->Body.MacAddr0, sizeof(wf200_mac_address_t));
147  memcpy(&(context->mac_addr_1.octet), startup_info->Body.MacAddr1, sizeof(wf200_mac_address_t));
148 
149  /* Sending to wf200 PDS configuration (Platform data set) contained in wf200_pds.h */
150  for ( uint8_t a = 0; a < ARRAY_COUNT(wf200_pds); a++ )
151  {
152  result = wf200_send_configuration( wf200_pds[a],strlen(wf200_pds[a]) );
153  }
154  ERROR_CHECK( result );
155  wf200_input_buffer_number = startup_info->Body.NumInpChBufs;
156 #ifdef DEBUG
157  printf("--PDS configured--\r\n");
158 #endif
159 
160 error_handler:
161  if ( result != SL_SUCCESS )
162  {
163  wf200_disable_irq( );
164  }
165 
166  return result;
167 }

398  }

1102  ERROR_CHECK( status );

1118  ERROR_CHECK( status );
1119 
1120  // skip signature and hash from image length
1121  image_length -= ( FW_HASH_SIZE + FW_SIGNATURE_SIZE );
1122

1124  num_blocks = (image_length - 1) / DOWNLOAD_BLOCK_SIZE + 1;
1125

1127  for ( block = 0; block < num_blocks; block++ )
1128  {
1129  /* check the download status in NCP */

1131  ERROR_CHECK( status );
1132

1134  {

1136  ERROR_CHECK( status );
1137  }
1138

1140  for ( i = 0; i < 100; i++ )
1141  {
1142  get = 0;
1143  status = wf200_apb_read_32( ADDR_DWL_CTRL_AREA_GET, &get );
1144  ERROR_CHECK( status );
1145 
1146  if ( ( put - get ) <= ( DOWNLOAD_FIFO_SIZE - DOWNLOAD_BLOCK_SIZE ) )
1147  {
1148  break;
1149  }
1150  }
1151 
1152  if ( ( put - get ) > ( DOWNLOAD_FIFO_SIZE - DOWNLOAD_BLOCK_SIZE ) )
1153  {
1154  status = SL_WIFI_FIRMWARE_DOWNLOAD_TIMEOUT;
1155  ERROR_CHECK( status );
1156  }
1157 
1158  /* calculate the block size */
1159  uint32_t block_size = image_length - put;
1160  if ( block_size > DOWNLOAD_BLOCK_SIZE )
1161  {
1162  block_size = DOWNLOAD_BLOCK_SIZE;
1163  }
1164 
1165  /* send the block to SRAM */
1166  status = wf200_host_get_firmware_data( &buffer, block_size );
1167  ERROR_CHECK( status );
1168  uint32_t block_address = ADDR_DOWNLOAD_FIFO_BASE + ( put % DOWNLOAD_FIFO_SIZE );
1169  status = wf200_apb_write( block_address, buffer, block_size );
1170  ERROR_CHECK( status );
1171 #ifdef DEBUG
1172  printf("Firmware page: %d/%d \n\r", put, image_length);
1173 #endif
1174  /* update the put register */
1175  put += block_size;
1176 
1177  status = wf200_apb_write_32( ADDR_DWL_CTRL_AREA_PUT, put );
1178  ERROR_CHECK( status );
1179 
1180  } /* End of firmware download loop */
1181 
1182  // notify NCP that upload ended
1183  status = wf200_apb_write_32( ADDR_DWL_CTRL_AREA_HOST_STATUS, HOST_STATE_UPLOAD_COMPLETE );
1184  ERROR_CHECK( status );
1185 
1186  // wait for authentication result
1187  status = poll_for_value( ADDR_DWL_CTRL_AREA_NCP_STATUS, NCP_STATE_AUTH_OK, 100 );
1188  ERROR_CHECK( status );
1189 
1190  // notify NCP that we are happy to run firmware
1191  status = wf200_apb_write_32( ADDR_DWL_CTRL_AREA_HOST_STATUS, HOST_STATE_OK_TO_JUMP );
1192  ERROR_CHECK( status );
1193 
1194 error_handler:
1195  wf200_host_deinit( );
1196 
1197  return status;
1198 }
