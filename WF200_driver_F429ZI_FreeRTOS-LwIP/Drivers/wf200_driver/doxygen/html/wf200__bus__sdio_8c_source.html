41 sl_status_t wf200_reg_read( wf200_register_address_t address, void* buffer, uint32_t length )
42 {
43  uint32_t buffer_id = 0;
44  uint32_t reg_address;
45  uint32_t current_transfer_size = ( length >= WF200_SDIO_BLOCK_MODE_THRESHOLD ) ? ROUND_UP( length, WF200_SDIO_BLOCK_SIZE ) : length;
46 
47  if ( address == WF200_IN_OUT_QUEUE_REG_ID )
48  {
49  buffer_id = rx_buffer_id++;
50  if ( rx_buffer_id > 4 )
51  {
52  rx_buffer_id = 1;
53  }
54  }
55 
56  reg_address = ( buffer_id << 7 ) | ( address << 2 );
57 
58  return wf200_host_sdio_transfer_cmd53( WF200_BUS_READ, 1, reg_address, buffer, current_transfer_size );
59 }

61 sl_status_t wf200_reg_write( wf200_register_address_t address, const void* buffer, uint32_t length )
62 {
63  uint32_t buffer_id = 0;
64  uint32_t reg_address;
65  uint32_t current_transfer_size = ( length >= WF200_SDIO_BLOCK_MODE_THRESHOLD ) ? ROUND_UP( length, WF200_SDIO_BLOCK_SIZE ) : length;
66 
67  if ( address == WF200_IN_OUT_QUEUE_REG_ID )
68  {
69  buffer_id = tx_buffer_id++;
70  if ( tx_buffer_id > 31 )
71  {
72  tx_buffer_id = 0;
73  }
74  }
75 
76  reg_address = ( buffer_id << 7 ) | ( address << 2 );
77 
78  return wf200_host_sdio_transfer_cmd53( WF200_BUS_WRITE, 1, reg_address, (void*)buffer, current_transfer_size );
79 }

82 sl_status_t wf200_init_bus( void )
83 {
84  sl_status_t result;
85  uint32_t value32;
86  uint8_t value_u8;
87 
88  rx_buffer_id = 1;
89  tx_buffer_id = 0;
90 
91  result = wf200_host_init_bus();
92  ERROR_CHECK(result);
93 
94  result = wf200_host_sdio_transfer_cmd52( WF200_BUS_READ, 0, 2, &value_u8 );
95  ERROR_CHECK( result );
96 
97  // Enables Function 1
98  value_u8 |= ( 1 << 1 );
99  result = wf200_host_sdio_transfer_cmd52( WF200_BUS_WRITE, 0, 2, &value_u8 );
100  ERROR_CHECK( result );
101 
102  // Enables Master and Function 1 interrupts
103  result = wf200_host_sdio_transfer_cmd52( WF200_BUS_READ, 0, 4, &value_u8 );
104  ERROR_CHECK( result );
105  value_u8 |= 0x1 | ( 1 << 1 );
106  result = wf200_host_sdio_transfer_cmd52( WF200_BUS_WRITE, 0, 4, &value_u8 );
107  ERROR_CHECK( result );
108 
109  // Set bus width to 4-bit
110  result = wf200_host_sdio_transfer_cmd52( WF200_BUS_READ, 0, 7, &value_u8 );
111  ERROR_CHECK( result );
112  value_u8 = ( value_u8 & 0xFC ) | 0x2;
113  result = wf200_host_sdio_transfer_cmd52( WF200_BUS_WRITE, 0, 7, &value_u8 );
114  ERROR_CHECK( result );
115 
116  // Switch to HS mode
117  result = wf200_host_sdio_transfer_cmd52( WF200_BUS_READ, 0, 0x13, &value_u8 );
118  ERROR_CHECK( result );
119  value_u8 |= 0x2; // Set EHS to 1
120  result = wf200_host_sdio_transfer_cmd52( WF200_BUS_WRITE, 0, 0x13, &value_u8 );
121  ERROR_CHECK(result);
122 
123  // Enabled SDIO high speed mode
124  wf200_host_sdio_enable_high_speed_mode( );
125 
126  // Set function 1 block size
127  value_u8 = WF200_SDIO_BLOCK_SIZE & 0xff;
128  result = wf200_host_sdio_transfer_cmd52( WF200_BUS_WRITE, 0, SDIO_FBR_FN1_BLOCK_SIZE_LSB_REGISTER, &value_u8 );
129  ERROR_CHECK( result );
130 
131  value_u8 = ( WF200_SDIO_BLOCK_SIZE >> 8 ) & 0xff;
132  result = wf200_host_sdio_transfer_cmd52( WF200_BUS_WRITE, 0, SDIO_FBR_FN1_BLOCK_SIZE_MSB_REGISTER, &value_u8 );
133  ERROR_CHECK( result );
134 
135  result = wf200_reg_read_32( WF200_CONFIG_REG_ID, &value32);
136  ERROR_CHECK(result);
137 
138  if ( value32 == 0 || value32 == 0xFFFFFFFF )
139  {
140  result = SL_ERROR;
141  ERROR_CHECK( result );
142  }
143 
144 error_handler:
145  return result;
146 }
