30 sl_status_t wf200_reg_read( wf200_register_address_t address, void* buffer, uint32_t length )
31 {
32  uint16_t header = SET_READ | ( address << 12 ) | ( length / 2 );
33 
34  uint8_t header_as_bytes[2];
35  PACK_16BIT_BIG_ENDIAN( header_as_bytes, header );
36 
37  wf200_host_spi_cs_assert( );
38  wf200_host_spi_transfer_no_cs_assert( WF200_BUS_WRITE, header_as_bytes, 2 );
39  wf200_host_spi_transfer_no_cs_assert( WF200_BUS_READ, buffer, length );
40  wf200_host_spi_cs_deassert( );
41 
42  if ( address == WF200_CONFIG_REG_ID )
43  {
44  // Config always read/written in "word mode 0"
45  // that is, its byte-order looks like that
46  // B1 B0 B3 B2
47  // here is this rearrangement
48  uint8_t* tmp = (uint8_t*) buffer;
49  uint8_t byte_swap = tmp[0];
50  tmp[0] = tmp[1];
51  tmp[1] = byte_swap;
52 
53  byte_swap = tmp[2];
54  tmp[2] = tmp[3];
55  tmp[3] = byte_swap;
56  }
57 
58  return SL_SUCCESS;
59 }

61 sl_status_t wf200_reg_write( wf200_register_address_t address, const void* buffer, uint32_t length )
62 {
63  if ( address == WF200_CONFIG_REG_ID )
64  {
65  // Config always read/written in "word mode 0"
66  // that is, its byte-order looks like that
67  // B1 B0 B3 B2
68  // here is this rearrangement
69  uint8_t* tmp = (uint8_t*) buffer;
70  uint8_t byte_swap = tmp[0];
71  tmp[0] = tmp[1];
72  tmp[1] = byte_swap;
73 
74  byte_swap = tmp[2];
75  tmp[2] = tmp[3];
76  tmp[3] = byte_swap;
77  }
78 
79  uint16_t header = /* write flag = 0*/( address << 12 ) | ( length / 2 );
80 
81  uint8_t header_as_bytes[2];
82  PACK_16BIT_BIG_ENDIAN( header_as_bytes, header );
83 
84  wf200_host_spi_cs_assert( );
85  wf200_host_spi_transfer_no_cs_assert( WF200_BUS_WRITE, header_as_bytes, 2 );
86  wf200_host_spi_transfer_no_cs_assert( WF200_BUS_READ, (uint8_t*)buffer, length ); // Note: wf200_host_spi_transfer_no_cs_assert() does not modify the buffer when doing SL_WIFI_BUS_WRITE
87  wf200_host_spi_cs_deassert( );
88 
89  return SL_SUCCESS;
90 }

93 sl_status_t wf200_init_bus( void )
94 {
95  sl_status_t status;
96  uint32_t value32;
97 
98  status = wf200_host_init_bus( );
99  ERROR_CHECK( status );
100 
101  wf200_host_reset_chip( );
102 
103  status = wf200_reg_read_32( WF200_CONFIG_REG_ID, &value32);
104  ERROR_CHECK(status);
105 
106  if(value32 == 0 || value32 == 0xFFFFFFFF)
107  {
108  status = SL_ERROR;
109  ERROR_CHECK(status);
110  }
111 
112  // setting up "word mode = b10", this means that data on SPI bus
113  // is sent as B0, B1, B2, B3 , ....
114  value32 &= ~WF200_CONFIG_WORD_MODE_BITS;
115  value32 |= WF200_CONFIG_WORD_MODE_2;
116 
117  /* Write CONFIG Register to configure SPI clock edge to support high frequency*/
118  value32 |= 0x04045400;
119 
120  status = wf200_reg_write_32( WF200_CONFIG_REG_ID, value32);
121  ERROR_CHECK( status );
122 
123  error_handler:
124  return status;
125 }
